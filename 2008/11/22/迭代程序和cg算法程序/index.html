迭代程序和CG算法程序

这几周老师总是说要交程序，虽然嘴上说有时间的话就做一做，没有时间的话就算了，可是说着了么长时间，听的都烦，结果周二的时候竟然说了一句，我看看那些人交了，心里有个底……听他这么一说，我心里倒是咯噔了一下，这个老师真阴险……于是这些天一直过的很窝火，总想找个时间把程序了结了，但是一想起来就很心烦，干什么事情都不能集中全力总是想着程序没写心里很不爽.今天呢其实很不爽，早晨起来都1点多了，其实不是睡懒觉，早就醒了，但是就是不想起来，想着起来后也没事情干还不如睡觉，但是睡觉又会觉得很浪费时间，所以总之因此更郁闷……下午去办公室把以前的作业稍微写了一下，六点多就回来准备写程序了.起点开始到现在终于搞定了~一下写了五个程序，感觉很有成就感,当然第一个是参考了别人的编程框架，之后的几个全是自己写的，嘿嘿心里的不爽终于释放了一些.写起来还是比较容易最后又经过了一个多小时的调试，终于完全弄好了~

以下代码在DEVc++中编译通过.

第一个是Jacobi的C 程序：

include<iostream>

include<math.h>

include<stdlib.h>

using namespace std;
const int m=10;
int main()
{

double a[m][m],b[m],e,x0[m],x1[m],w,se,max;
int n,i,j,N,k;
N=10000;e=0.000001;
cout<<”———————————————————————————-“<<endl;
cout<<”n请输入方程的个数：”;
cin>>n;cout<<endl;
for(i=1;i<=n;i++)
{
cout<<”请输入第”<<i<<”个方程的各项系数：”;
for(j=1;j<=n;j++)
cin>>a[i][j];
}
cout<<”n请输入各个方程等号右边的常数项:n”;
for(i=1;i<=n;i++)
{
cin>>b[i];
}
for(i=1;i<=n;i++)
{
x0[i]=0;
x1[i]=x0[i];
}
for(k=1;k<=N;k++)
{
for(i=1;i<=n;i++) //求x(k+1)
{
w=0;
for(j=1;j<=n;j++)
{
if(j!=i)
w=w+a[i][j]*x0[j];
}
x1[i]=(b[i]-w)/double(a[i][i]);
}

max=fabs(x0[1]-x1[1]); //求残量范数 ||x(k+1)-x(k)||/||x(k+1)||;
for(i=1;i<=n;i++)
{
se=fabs(x0[i]-x1[i]);
if(se>max)
max=se;
}
if(max<e) //判断迭代是否收敛
{
cout<<endl;
for(i=1;i<=n;i++)
cout<<”x”<<i<<”=”<<x1[i]<<endl;
cout<<endl;
cout<<”迭代次数”<<”:”<<k;
break;
}
for(i=1;i<=n;i++) //赋值，重新计算
{
x0[i]=x1[i];
}
}
if(k==N)
cout<<”迭代失败！！”<<endl;
cout<<endl;
cout<<”———————————————————————————-“<<endl;
cout<<endl;
system(“pause”);
return 0;
}

include<math.h>

include<stdlib.h>

using namespace std;
const int m=10;
int main()
{

double a[m][m],b[m],alfa,beta,p[m],rou1,rou2,e,c,x[m],r[m],w,se,max;
int n,i,j,N,k;
N=10000;e=0.000001;
cout<<”———————————————————————————-“<<endl;
cout<<”n请输入方程的个数：”;
cin>>n;cout<<endl;
for(i=1;i<=n;i++)
{
cout<<”请输入第”<<i<<”个方程的各项系数：”;
for(j=1;j<=n;j++)
cin>>a[i][j];
}
cout<<”n请输入各个方程等号右边的常数项：n”;
for(i=1;i<=n;i++)
{
cin>>b[i];
}
cout<<”n请输入x的初始值：取（0，0，0）迭代效果比较好n”;
for(i=1;i<=n;i++)
{
cin>>x[i];
}
for(i=1;i<=n;i++) //计算r0
{
for(j=1;j<=n;j++)
{
r[i]=b[i]-a[i][j]x[j];
}
}
max=fabs(r[1]);
for(i=1;i<=n;i++)
{
se=fabs(r[i]);
if(se>max)
max=se;
}
if(max<e)
{
for(i=1;i<=n;i++)
cout<<”x”<<i<<”=”<<x[i]<<endl;
cout<<endl;
}
else
{
rou1=0;
for(i=1;i<=n;i++)
{
p[i]=r[i];
rou1=rou1+r[i]r[i];
}
}
for(k=1;k<=N;k++) //迭代
{
c=0;
for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
{
c=c+p[i]a[i][j]p[j];
}
}

for(i=1;i<=n;i++)
{
x[i]=x[i]+alfap[i];
}
for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
{
r[i]=r[i]-alfaa[i][j]p[j];
}
}
rou2=0;
for(i=1;i<=n;i++)
{
rou2=rou2+r[i]r[i];
}
beta=rou2/rou1;
for(i=1;i<=n;i++)
{
p[i]=r[i]+beta*p[i];
}
rou1=rou2;
